from flask import Flask, jsonify, request import os import sqlite3 app = Flask(__name__) # ========================== # DATABASE # ========================== db = sqlite3.connect("database.db", check_same_thread=False) cursor = db.cursor() cursor.execute(""" CREATE TABLE IF NOT EXISTS settings ( guild_id INTEGER PRIMARY KEY, anti_profanity INTEGER DEFAULT 1 ) """) db.commit() # ========================== # HOME PAGE # ========================== @app.route("/") def home(): return "🟢 Bot Render üzerinde çalışıyor!" # ========================== # GET SETTINGS # ========================== @app.route("/api/settings/") def get_settings(guild_id): cursor.execute( "SELECT anti_profanity FROM settings WHERE guild_id=?", (guild_id,) ) data = cursor.fetchone() if data is None: return jsonify({ "guild_id": guild_id, "anti_profanity": 1 }) return jsonify({ "guild_id": guild_id, "anti_profanity": data[0] }) # ========================== # TOGGLE PROFANITY # ========================== @app.route("/api/toggle/profanity/", methods=["POST"]) def toggle_profanity(guild_id): cursor.execute( "SELECT anti_profanity FROM settings WHERE guild_id=?", (guild_id,) ) data = cursor.fetchone() # Eğer yoksa oluştur if data is None: cursor.execute( "INSERT INTO settings VALUES (?, ?)", (guild_id, 0) ) db.commit() return jsonify({ "guild_id": guild_id, "anti_profanity": 0, "status": "created" }) # tersine çevir new_value = 0 if data[0] == 1 else 1 cursor.execute( "REPLACE INTO settings VALUES (?, ?)", (guild_id, new_value) ) db.commit() return jsonify({ "guild_id": guild_id, "anti_profanity": new_value, "status": "updated" }) # ========================== # RUN FUNCTION (RENDER UYUMLU) # ========================== def run(): port = int(os.environ.get("PORT", 5000)) app.run(host="0.0.0.0", port=port)
